📥 IR Code Capture - Arduino Code

#include <IRremote.h>
int recvPin = 15;
IRrecv irrecv(recvPin);
decode_results results;
void setup() { Serial.begin(9600); irrecv.enableIRIn(); }
void loop() { if (irrecv.decode(&results)) { Serial.println(results.value, HEX); irrecv.resume(); } }

Sample Codes: Power: 0xFFA25D, Vol+: 0xFF629D

📤 IR Signal Send - Arduino Code

#include <IRremote.h>
IRsend irsend;
void loop() { irsend.sendNEC(0xFFA25D, 32); delay(5000); }

📊 Sensor Data Reading

#include <DHT.h>
DHT dht(4, DHT11);
int pirPin = 14;
void loop() {
  float temp = dht.readTemperature();
  int motion = digitalRead(pirPin);
  Serial.print(temp); Serial.print("°C | Motion: "); Serial.println(motion);
  delay(2000);
}

⚡ Device Control via Relay

int relayPin = 5;
void setup() { pinMode(relayPin, OUTPUT); }
void loop() {
  if(Serial.available()) {
    char cmd = Serial.read();
    digitalWrite(relayPin, cmd=='1'?HIGH:LOW);
  }
}
